Quick Start

// Initialize the BusinessAi API client
const BusinessAi = require('businessai-sdk');

const client = new BusinessAi({
  apiKey: 'your_api_key_here',
  environment: 'production' // or 'sandbox'
});

// Example API call for market analysis
client.marketAnalysis.predict({
  symbol: 'AAPL',
  timeframe: '1d',
  indicators: ['RSI', 'MACD']
})
.then(response => {
  console.log('Prediction results:', response.data);
})
.catch(error => {
  console.error('API Error:', error);
});
